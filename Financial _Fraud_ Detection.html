import shap
# Sample the test set once to ensure consistency
sample_X = X_test.sample(1000, random_state=42)
# Create SHAP explainer
explainer = shap.TreeExplainer(model)
# Get SHAP values
shap_values = explainer.shap_values(sample_X)
# If shap_values is a list (binary classification), use the class 1 explanation
if isinstance(shap_values, list):
    shap.summary_plot(shap_values[1], sample_X, plot_type="bar")
else:
    shap.summary_plot(shap_values, sample_X, plot_type="bar")
